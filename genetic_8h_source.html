81 T selectMember()
82 {
83 // totalFitness could be 0, so we have to max with 1 to avoid % 0

85 int randomNum = std::max(rand() % std::max(int(totalFitness), 1), 1);

100 float biasedFitness(T member)
101 {
102 int sign = 1;
103 float f = member.fitness();
104
105 if (f < 0)
106 {
107 f *= -1;
108 sign *= -1;
109 }
110
111 float r = sign * pow(f, bias);
112
113 return max ? r : 1 / r; // If we're minning we need to invert r to pick the best ones
114 }
